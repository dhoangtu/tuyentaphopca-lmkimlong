graceOn =
\applyContext
#(lambda (c)
  (for-each
   (lambda (e)
    (let ((ct (ly:context-find c (car e))))
     (if ct
      (apply ly:context-pushpop-property ct (cdr e)))))
   (ly:context-property c 'graceSettings)))

graceOff =
\applyContext
#(lambda (c)
  (for-each
   (lambda (e)
    (let ((ct (ly:context-find c (car e))))
     (if ct
      (apply ly:context-pushpop-property ct (list-head (cdr e) 2)))))
   (ly:context-property c 'graceSettings)))
